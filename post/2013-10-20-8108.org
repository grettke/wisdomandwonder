#+BEGIN_EXAMPLE
    
(defun gcr/plist-to-alist (ls)
  "Convert a plist to an alist. Primarily for old color-theme themes."
  (let ((result nil))
    (while ls
      (add-to-list 'result (cons (car ls) (cadr ls)))
      (setq ls (cddr ls)))
    result))

(defalias 'plist-to-alist 'gcr/plist-to-alist)

#+END_EXAMPLE
